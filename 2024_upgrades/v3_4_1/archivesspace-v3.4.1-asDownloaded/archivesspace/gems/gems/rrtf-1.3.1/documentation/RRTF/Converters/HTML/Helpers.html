def font(key)
  RRTF::Font.new(*case key
    when :default   then [RTF::Font::ROMAN,  'Times New Roman']
    when :monospace then [RTF::Font::MODERN, 'Courier New'    ]
  end)
end

def style(key)
  RRTF::CharacterStyle.new.tap do |style|
    case key.to_sym
    when :h1
      style.font_size = 44
      style.bold = true
    when :h2
      style.font_size = 36
      style.bold = true
    when :h3
      style.font_size = 28
      style.bold = true
    when :h4
      style.font_size = 22
      style.bold = true
    end
  end
end
