Overview

This class represents the font table for an RTF document. An instance of the class is used internally by the Document class and should not need to be explicitly instantiated (although it can be obtained from a Document object if needed).

def add(font)
   if font.instance_of?(Font)
      @fonts.push(font) if @fonts.index(font).nil?
   end

def to_rtf(indent=0)
   prefix = indent > 0 ? ' ' * indent : ''
   text   = StringIO.new
   text << "#{prefix}{\\fonttbl"
   @fonts.each_index do |index|
      text << "\n#{prefix}{\\f#{index}#{@fonts[index].to_rtf}}"
   end
   text << "\n#{prefix}}"
   text.string
end

def to_s(indent=0)
   prefix = indent > 0 ? ' ' * indent : ''
   text   = StringIO.new
   text << "#{prefix}Font Table (#{@fonts.size} fonts)"
   @fonts.each {|font| text << "\n#{prefix}   #{font}"}
   text.string
end
